Javascript to Build Table from Array

var table = document.getElementById("convertTable");
for (var y=0; y < tableArray.length; y++) {
 var tr = document.createElement("tr");
 for (var x=0; x < tableArray[0].length; x++) {
  var td = document.createElement("td");
  var text = document.createTextNode(tableArray[y][x]);
  td.appendChild(text);
  tr.appendChild(td);
 }
 table.appendChild(tr);
}
